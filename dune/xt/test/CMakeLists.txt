dependencycheck(${xtcommon})

  gtest/gtest-all.cxx)

        DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}/dune/xt/test/

        PATTERN "*")

add_subdir_tests(common)
add_subdir_tests(grid)
add_subdir_tests(la)
add_subdir_tests(functions)
